**Card grid:**
```tsx

  initial="hidden"
  whileInView="visible"

  className="grid grid-cols-1 md:grid-cols-3 gap-6"
>
  {differentiators.map(({ Icon, headline, expansion }) => (
    <motion.div
      key={headline}
      variants={itemVariants}

</motion.div>
```
